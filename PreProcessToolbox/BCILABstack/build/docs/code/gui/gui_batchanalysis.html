0328 function data = resolve_data(data,name)
0329 % resolve all references to data in the given cell array of strings
0330 % allowed formats are:
0331 % - varname or quoted string, or unquoted file name
0332 % - cell array of varnames or quoted strings (but not of file names)
0333 % for each cell...
0334 try
0335     for c=1:length(data)
0336         data{c} = strtrim(data{c});
0337         % check if this is an unquoted file name reference
0338         if (data{c}(1) ~= '{' || ~data{c}(end) ~= '}') && (~data{c}(1) ~= '''' || ~data{c}(end) ~= '''')
0339             if exist(data{c},'file')
0340                 % it's a file name: wrap in quotes
0341                 data{c} = ['''' data{c} ''''];
0342             elseif (any(data{c}=='*') || any(data{c}=='?')) && ~isempty(dir(data{c}))
0343                 % it's apparently a file name pattern: wrap in quotes
0344                 data{c} = ['''' data{c} ''''];
0345             elseif isvarname(data{c}) && evalin('base',sprintf('exist(''%s'',''var'')',data{c}))
0346                 % assume it's a varname
0347             else
0348                 error(['The follwoing unqouted data set refers neither to a file on disk nor a variable in the workspace: ' data{c}]);
0349             end
0350         end
0351         
0352         % try to evaluate it
0353         try
0354             data{c} = evalin('base',data{c});
0355         catch e
0356             env_handleerror(e);
0357             error(['The following data set reference cannot be resolved: ' data{c}]);
0358         end
0359     end
0360 catch e
